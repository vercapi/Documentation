*** Preview a theme

#+name: preview
#+header: :var theme='debian-moreblue'
#+header: :var path='/usr/share/slim/themes'
#+BEGIN_SRC sh
slim -p $path/$theme
#+END_SRC

#+RESULTS: preview

#+CALL: preview(theme='arch-tetra')

#+RESULTS:
: nil


*** Set a theme

Install the theme in /usr/share/slim/themes

Change /etc/slim.conf
current_theme <name-of-the-theme-folder>
